view :updated_by, wrap: { div: { class: "text-muted" } } do
  updaters = Card.search(updater_of: { id: card.id }) if card.id
  return "" unless updaters.present?

  links = updater_links updaters, others_target: card.fetch(:editors)
  "Updated by #{links}"
end
